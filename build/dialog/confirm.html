1.基本调用

点击这里弹出
seajs.use(['comfirmbox'], function(ConfirmBox) {
    var cb = new ConfirmBox({
        trigger: '#J-confirm-dialog',
        title: '这里是标题',
        width:300,
        message: '这里是内容',
        onConfirm: function() {
            var that = this;
            alert("回调写在onConfirm里面");
            that.hide();
        }
    });
});

2.ConfirmBox静态方法---ConfirmBox.alert

信息提示 成功信息 警告信息 错误信息 复杂提示信息 
 seajs.use(['$','comfirmbox'],function($,Dialog){
    $("#J-alert-info").click(function(e){
        e.preventDefault();
        ConfirmBox.alert('这是个普通的提示',null,{width:300,stype:'info'});
    });

    $("#J-alert-success").click(function(e){
        e.preventDefault();
        ConfirmBox.alert('这是成功的提示',null,{width:300,stype:'success'});
    });

    $("#J-alert-warn").click(function(e){
        e.preventDefault();
        ConfirmBox.alert('这是警告提示',null,{width:300,stype:'warn'});
    });

        $("#J-alert-error").click(function(e){
        e.preventDefault();
        ConfirmBox.alert('这里是错误的提示',null,{width:300,stype:'error'});
    });

    $("#J-alert-fz").click(function(e){
        e.preventDefault();
        var _errorHtml = '上传失败,请上传10M以内的文件<br>';
            _errorHtml += '<span class="gray f12">有问题请咨询:0571-34566777</span><br>'
            _errorHtml += '<span class="f12"><a href="#">返回首页</></span>';
        ConfirmBox.alert(_errorHtml,null,{width:500,stype:'error'});
    });
 });
