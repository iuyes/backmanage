4.内容传入url

点击这里
seajs.use(['$','dialog'],function($,Dialog){

    new Dialog({
        trigger: '#J-url-dialog',
       content: '<div class="dialog_loading"></div>'
    }).before('show',function(){
        var url = $("#J-url-dialog").attr('href');
        this.set('content',url)
    }); 
  });
